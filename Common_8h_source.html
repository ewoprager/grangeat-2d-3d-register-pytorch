122template <typename T> [[nodiscard]] __host__ __device__ Vec<T, 3> UnflipSphericalCoordinate(const Vec<T, 3> &coordSph) {
123 constexpr T PI_T = static_cast<T>(M_PI);
124 const T theta_div = std::floor((coordSph.Y() + .5f * PI_T) / PI_T);
125 const bool theta_flip = static_cast<bool>(std::abs(static_cast<long>(theta_div)) % 2);
126 const T phi_div = std::floor((coordSph.Z() + .5f * PI_T) / PI_T);
127 const bool phi_flip = static_cast<bool>(std::abs(static_cast<long>(phi_div)) % 2);
128
129 T r_ret = coordSph.X();
130 T theta_ret = coordSph.Y() - PI_T * theta_div;
131 T phi_ret = coordSph.Z() - PI_T * phi_div;
132
133 if (phi_flip && !theta_flip) theta_ret *= -1;
134 if (phi_flip != theta_flip) r_ret *= -1;
135
136 return {r_ret, theta_ret, phi_ret};
137}
