118template <typename T> [[nodiscard]] __host__ __device__ Vec<T, 3> UnflipSphericalCoordinate(const Vec<T, 3> &coordSph) {
119 constexpr T PI_T = static_cast<T>(M_PI);
120 const T theta_div = std::floor((coordSph.Y() + .5f * PI_T) / PI_T);
121 const bool theta_flip = static_cast<bool>(std::abs(static_cast<long>(theta_div)) % 2);
122 const T phi_div = std::floor((coordSph.Z() + .5f * PI_T) / PI_T);
123 const bool phi_flip = static_cast<bool>(std::abs(static_cast<long>(phi_div)) % 2);
124
125 T r_ret = coordSph.X();
126 T theta_ret = coordSph.Y() - PI_T * theta_div;
127 T phi_ret = coordSph.Z() - PI_T * phi_div;
128
129 if (phi_flip && !theta_flip) theta_ret *= -1;
130 if (phi_flip != theta_flip) r_ret *= -1;
131
132 return {r_ret, theta_ret, phi_ret};
133}
