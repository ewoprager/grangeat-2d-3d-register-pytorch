84 __host__ static CommonData Common(const at::Tensor &image, const at::Tensor &imageSpacing,
85 const at::Tensor &phiValues, const at::Tensor &rValues, int64_t samplesPerLine,
86 at::DeviceType device) {
87 // image should be a 2D tensor of floats on the chosen device
88 TORCH_CHECK(image.sizes().size() == 2);
89 TORCH_CHECK(image.dtype() == at::kFloat);
90 TORCH_INTERNAL_ASSERT(image.device().type() == device);
91 // imageSpacing should be a 1D tensor of 2 floats or doubles
92 TORCH_CHECK(imageSpacing.sizes() == at::IntArrayRef{2});
93 TORCH_CHECK(imageSpacing.dtype() == at::kFloat || imageSpacing.dtype() == at::kDouble);
94 // phiValues and rValues should have the same sizes, should contain floats, and be on the chosen device
95 TORCH_CHECK(phiValues.sizes() == rValues.sizes());
96 TORCH_CHECK(phiValues.dtype() == at::kFloat);
97 TORCH_CHECK(rValues.dtype() == at::kFloat);
98 TORCH_INTERNAL_ASSERT(phiValues.device().type() == device);
99 TORCH_INTERNAL_ASSERT(rValues.device().type() == device);
100
101 CommonData ret{};
102 ret.inputTexture = texture_t::FromTensor(image, imageSpacing);
103 const double lineLength = sqrt(ret.inputTexture.SizeWorld().template Apply<double>(&Square<double>).Sum());
104 ret.mappingIndexToOffset = GetMappingIToOffset(lineLength, samplesPerLine);
105 ret.scaleFactor = lineLength / static_cast<float>(samplesPerLine);
106 ret.flatOutput = torch::zeros(at::IntArrayRef({phiValues.numel()}), image.contiguous().options());
107 return ret;
108 }

157 __host__ __device__ [[nodiscard]] static Vec<double, 2> GetDTexCoordDR(const texture_t &textureIn, double phi,
158 double r) {
159 const double sign = static_cast<double>(r > 0.) - static_cast<double>(r < 0.);
160 const double s = sin(phi);
161 const double c = cos(phi);
162 return textureIn.MappingWorldToTexCoord().gradient * sign * Vec<double, 2>{c, s};

177 __host__ __device__ [[nodiscard]] static float IntegrateLooped(const texture_t &texture,
178 const Linear<Vec<double, 2>> &mappingIndexToTexCoord,
179 int64_t samplesPerLine) {
180 float ret = 0.f;
181 for (int64_t i = 0; i < samplesPerLine; ++i) {
182 const double iF = static_cast<double>(i);
183 ret += texture.Sample(mappingIndexToTexCoord(Vec<double, 2>::Full(iF)));
184 }
185 return ret;

198 __host__ __device__ [[nodiscard]] static float
199 DIntegrateLoopedDMappingParameter(const texture_t &texture, const Linear<Vec<double, 2>> &mappingIndexToTexCoord,
200 const Vec<double, 2> &dTexCoordDMappingParameter, int64_t samplesPerLine) {
201 float ret = 0.f;
202 for (int64_t i = 0; i < samplesPerLine; ++i) {
203 const double iF = static_cast<double>(i);
204 const Vec<double, 2> texCoord = mappingIndexToTexCoord(Vec<double, 2>::Full(iF));
205 ret += texture.DSampleDX(texCoord) * dTexCoordDMappingParameter.X() +
206 texture.DSampleDY(texCoord) * dTexCoordDMappingParameter.Y();
207 }
208 return ret;
209 }
