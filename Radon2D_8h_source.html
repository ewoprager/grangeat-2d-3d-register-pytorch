92 __host__ static CommonData Common(const at::Tensor &image, const at::Tensor &imageSpacing,
93 const at::Tensor &phiValues, const at::Tensor &rValues, int64_t samplesPerLine,
94 at::DeviceType device) {
95 // image should be a 2D tensor of floats on the chosen device
96 TORCH_CHECK(image.sizes().size() == 2);
97 TORCH_CHECK(image.dtype() == at::kFloat);
98 TORCH_INTERNAL_ASSERT(image.device().type() == device);
99 // imageSpacing should be a 1D tensor of 2 doubles
100 TORCH_CHECK(imageSpacing.sizes() == at::IntArrayRef{2});
101 TORCH_CHECK(imageSpacing.dtype() == at::kDouble);
102 // phiValues and rValues should have the same sizes, should contain floats, and be on the chosen device
103 TORCH_CHECK(phiValues.sizes() == rValues.sizes());
104 TORCH_CHECK(phiValues.dtype() == at::kFloat);
105 TORCH_CHECK(rValues.dtype() == at::kFloat);
106 TORCH_INTERNAL_ASSERT(phiValues.device().type() == device);
107 TORCH_INTERNAL_ASSERT(rValues.device().type() == device);
108
109 CommonData ret{};
110 ret.inputTexture = texture_t::FromTensor(image, imageSpacing);
111 const double lineLength = sqrt(ret.inputTexture.SizeWorld().template Apply<double>(&Square<double>).Sum());
112 ret.mappingIndexToOffset = GetMappingIToOffset(lineLength, samplesPerLine);
113 ret.scaleFactor = lineLength / static_cast<float>(samplesPerLine);
114 ret.flatOutput = torch::zeros(at::IntArrayRef({phiValues.numel()}), image.contiguous().options());
115 return ret;
116 }

129 __host__ __device__ [[nodiscard]] static Linear<Vec<double, 2> > GetMappingIToOffset(double lineLength,
130 int64_t samplesPerLine) {
131 return {Vec<double, 2>::Full(-.5f * lineLength),

145 __host__ __device__ [[nodiscard]] static Linear<Vec<double, 2> >
146 GetMappingIndexToTexCoord(const texture_t &textureIn, double phi, double r,
147 const Linear<Vec<double, 2> > &mappingIToOffset) {
148 const double s = sin(phi);
149 const double c = cos(phi);
150 const Linear<Vec<double, 2> > mappingOffsetToWorld{{r * c, r * s}, {-s, c}};
151 return textureIn.MappingWorldToTexCoord()(mappingOffsetToWorld(mappingIToOffset));

165 __host__ __device__ [[nodiscard]] static Vec<double, 2> GetDTexCoordDR(const texture_t &textureIn, double phi,
166 double r) {
167 const double sign = static_cast<double>(r > 0.) - static_cast<double>(r < 0.);
168 const double s = sin(phi);
169 const double c = cos(phi);
170 return textureIn.MappingWorldToTexCoord().gradient * sign * Vec<double, 2>{c, s};
171 }

185 __host__ __device__ [[nodiscard]] static float IntegrateLooped(const texture_t &texture,
186 const Linear<Vec<double, 2> > &
187 mappingIndexToTexCoord, int64_t samplesPerLine) {
188 float ret = 0.f;

206 __host__ __device__ [[nodiscard]] static float
207 DIntegrateLoopedDMappingParameter(const texture_t &texture, const Linear<Vec<double, 2> > &mappingIndexToTexCoord,
208 const Vec<double, 2> &dTexCoordDMappingParameter, int64_t samplesPerLine) {
209 float ret = 0.f;
210 for (int64_t i = 0; i < samplesPerLine; ++i) {
211 const double iF = static_cast<double>(i);
212 const Vec<double, 2> texCoord = mappingIndexToTexCoord(Vec<double, 2>::Full(iF));
213 ret += texture.DSampleDX(texCoord) * dTexCoordDMappingParameter.X() + texture.DSampleDY(texCoord) *
214 dTexCoordDMappingParameter.Y();
215 }
216 return ret;
