81 [[nodiscard]] __host__ __device__ float Sample(VectorType rThetaPhi) const {
82 const FloatType z = sin(rThetaPhi.Y());
83 const FloatType nSideF = static_cast<FloatType>(nSide);
84 const FloatType i = nSideF * (2.0 - 1.5 * z);
85 FloatType jEquiv;
86 if (z > 0.666666666666666667) {
87 // HEALPix top polar cap
88 const FloatType j = 2.0 * i * rThetaPhi.Z() / M_PI + 0.5;
89 jEquiv = nSideF * floor((j - 1.0) / i) + 1.0 + floor(0.5 * (nSideF - 1.0)) + fmod(j - 1.0, i);
90 } else if (z > -0.66666666666667) {
91 // HEALPix equatorial zone
92 const FloatType j = 2.0 * nSideF * rThetaPhi.Z() / M_PI + 0.5 * fmod(i - nSideF + 1.0, 2.0);
93 jEquiv = j;
94 } else {
95 // HEALPix bottom polar cap
96 const FloatType j = 2.0 * (4.0 * nSideF - i) * rThetaPhi.Z() / M_PI + 0.5;
97 jEquiv = nSideF * floor((j - 1.0) / (4.0 * nSideF - i)) + 1.0 + floor(0.5 * (nSideF - 1.0)) + fmod(
98 j - 1.0, 4.0 * nSideF - i);
99 }
100 const FloatType deltaV48 = jEquiv - floor(0.5 * i) + 1.0 < 0.0 ? nSideF : 0.0;
101 // r,u,v is the order of the texture dimensions
102 const VectorType ruv = {rThetaPhi.X(), fmod(jEquiv - floor(0.5 * i) + 1.0, 4.0 * nSideF),
103 fmod(jEquiv + floor(0.5 * (i + 1.0)) - 3.0 + deltaV48, 3.0 * nSideF)};
104 // texCoord is in the reverse order: (X, Y, Z)
105 const VectorType texCoord = ruv.Flipped() * Size().StaticCast<double>() - .5;
106 const VectorType floored = texCoord.Apply<double>(&floor);
107 const SizeType index = floored.StaticCast<int64_t>();
108 const VectorType fractions = texCoord - floored;
109 const float l0r0 = (1.f - fractions.X()) * At(index) + fractions.X() *
110 At({index.X() + 1, index.Y(), index.Z()});
111 const float l0r1 = (1.f - fractions.X()) * At({index.X(), index.Y() + 1, index.Z()}) + fractions.X() * At(
112 {index.X() + 1, index.Y() + 1, index.Z()});
113 const float l1r0 = (1.f - fractions.X()) * At({index.X(), index.Y(), index.Z() + 1}) + fractions.X() * At(
114 {index.X() + 1, index.Y(), index.Z() + 1});
115 const float l1r1 = (1.f - fractions.X()) * At({index.X(), index.Y() + 1, index.Z() + 1}) + fractions.X() * At(
116 {index.X() + 1, index.Y() + 1, index.Z() + 1});
117 const float l0 = (1.f - fractions.Y()) * l0r0 + fractions.Y() * l0r1;
118 const float l1 = (1.f - fractions.Y()) * l1r0 + fractions.Y() * l1r1;
119 return (1.f - fractions.Z()) * l0 + fractions.Z() * l1;
120 }
