215    def grid_sample_smoothed(self, grid: Sinogram3dGrid, *, i_mapping: LinearMapping, j_mapping: LinearMapping,
216                             k_mapping: LinearMapping, sigma: float, offset_count: int = 10):
217        
218        Sample the sinogram at the given phi, theta, r spherical coordinates, with extra samples in a Gaussian layout
219        around the sampling positions to make the sampling more even over S^2, even if the point distribution is less
220        even.
221
222        :param grid: A grid of 3D sinogram coordinates
223        :param i_mapping: Mapping from r to sinogram texture x-coordinate (-1, 1)
224        :param j_mapping: Mapping from theta to sinogram texture y-coordinate (-1, 1)
225        :param k_mapping: Mapping from phi to sinogram texture z-coordinate (-1, 1)
226        :param offset_count: Number of rows and columns of offset points to make weighted samples at
227        :param sigma: The standard deviation of the Gaussian pattern
228        :return: A tensor matching size of `phi_values`  - the weighted sums of offset samples around the given
229        coordinates.
230        
231
232        assert grid.phi.device == self.device
233        assert grid.device_consistent()
234        assert grid.size_consistent()
235        assert sigma >= 0.
236
237        logger.info("Sample smoothing with sigma = {:.3f}".format(sigma))
238
239        # Gaussian weighed sampling offsets
240        xs = torch.linspace(-3. * sigma, 3. * sigma, offset_count)
241        ys = torch.linspace(-3. * sigma, 3. * sigma, offset_count)
242        ys, xs = torch.meshgrid(ys, xs)
243        xs = xs.flatten()
244        ys = ys.flatten()
245        weights = (-(xs.square() + ys.square()) / (2. * sigma * sigma)).exp()
246        weights = weights / weights.sum()
247
248        # New offset values of phi & theta are determined by rotating the vector (1, 0, 0)^T first by a small
249        # perturbation
250        # according to the Gaussian pattern, and then by the original rotation according to the old values of phi *
251        # theta.
252
253        # Determining a perturbed vector for each offset in the Gaussian pattern:
254        phis_off = torch.atan2(ys, xs)
255        thetas_off = (xs.square() + ys.square()).sqrt()
256
257        del xs, ys
258
259        cp = phis_off.cos()
260        sp = phis_off.sin()
261        ct = thetas_off.cos()
262        st = thetas_off.sin()
263
264        del phis_off, thetas_off
265
266        offset_vectors = torch.stack((ct, -st * sp, st * cp), dim=-1).to(device=self.device)
267
268        # Determining the rotation matrices for each input (phi, theta):
269        cp = grid.phi.cos()
270        sp = grid.phi.sin()
271        ct = grid.theta.cos()
272        st = grid.theta.sin()
273        row_0 = torch.stack((cp * ct, -sp, -cp * st), dim=-1)
274        row_1 = torch.stack((sp * ct, cp, sp * st), dim=-1)
275        row_2 = torch.stack((st, torch.zeros_like(st), ct), dim=-1)
276        rotation_matrices = torch.stack((row_0, row_1, row_2), dim=-2).to(device=self.device)
277        # Multiplying by the perturbed unit vectors for the perturbed, rotated unit vector:
278        rotated_vectors = torch.einsum('...ij,kj->...ki', rotation_matrices, offset_vectors)
279
280        del cp, sp, ct, st, row_0, row_1, row_2, rotation_matrices, offset_vectors
281
282        # Converting the resulting unit vectors back into new values of phi & theta, and expanding the r tensor to match
283        # in size:
284        new_phis = torch.atan2(rotated_vectors[..., 1], rotated_vectors[..., 0])
285        new_thetas = torch.clamp(rotated_vectors[..., 2], -1., 1.).asin()
286
287        del rotated_vectors
288
289        new_grid = Sinogram3dGrid(
290            new_phis, new_thetas, grid.r.unsqueeze(-1).expand(
291                [-1] * len(grid.r.size()) + [new_phis.size()[-1]]).clone()).unflip()  # need to
292        # clone the r grid otherwise it's just a tensor view, not a tensor in its own right
293
294        del new_phis, new_thetas
295
296        # Sampling at all the perturbed orientations:
297        grid = torch.stack((i_mapping(new_grid.r), j_mapping(new_grid.theta), k_mapping(new_grid.phi)), dim=-1)
298        samples = reg23.grid_sample3d(self.data, grid, address_mode="wrap")
299
300        del grid
301
302        ##
303        _, axes = plt.subplots()
304        mesh = axes.pcolormesh(torch.einsum('i,...i->...', weights, new_grid.phi.cpu()).cpu())
305        axes.axis('square')
306        axes.set_title("average phi_sph resampling values")
307        axes.set_xlabel("r_pol")
308        axes.set_ylabel("phi_pol")
309        plt.colorbar(mesh)
310        _, axes = plt.subplots()
311        mesh = axes.pcolormesh(torch.einsum('i,...i->...', weights, new_grid.theta.cpu()).cpu())
312        axes.axis('square')
313        axes.set_title("average theta_sph resampling values")
314        axes.set_xlabel("r_pol")
315        axes.set_ylabel("phi_pol")
316        plt.colorbar(mesh)
317        _, axes = plt.subplots()
318        mesh = axes.pcolormesh(torch.einsum('i,...i->...', weights, new_grid.r.cpu()).cpu())
319        axes.axis('square')
320        axes.set_title("average r_sph resampling values")
321        axes.set_xlabel("r_pol")
322        axes.set_ylabel("phi_pol")
323        plt.colorbar(mesh)
324        ##
325
326        # Applying the weights and summing along the last dimension for an output equal in size to the input tensors of
327        # coordinates:
328        return torch.einsum('i,...i->...', weights.to(device=self.device), samples)
