 18def to_latex_scientific(x: float, precision: int = 2, include_plus: bool = False):
 19    if x == 0:
 20        return f"{0:.{precision}f}"
 21    exponent: int = int(f"{x:e}".split("e")[1])
 22    mantissa: float = x / (10.0 ** exponent)
 23    if exponent == 0:
 24        return f"{mantissa:.{precision}f}"
 25    if include_plus:
 26        return fr"{mantissa:+.{precision}f} \times 10^{{{exponent}}}"
 27    return fr"{mantissa:.{precision}f} \times 10^{{{exponent}}}"

 30def torch_polyfit(xs: torch.Tensor, ys: torch.Tensor, *, degree: int = 1) -> torch.Tensor:
 31    assert xs.size() == ys.size()
 32
 33    A = torch.stack([xs.pow(i) for i in range(degree + 1)], dim=1)  # size = (n, degree + 1)
 34    B = ys.unsqueeze(1)
 35
 36    ret = torch.linalg.lstsq(A, B)
 37
 38    return ret.solution

 41def fit_power_relationship(xs: torch.Tensor, ys: torch.Tensor) -> torch.Tensor:
 42    """
 43    y = a * x^b
 44    :param xs:
 45    :param ys:
 46    :return: (2,) tensor containing [a, b]
 47    """
 48    assert xs.size() == ys.size()
 49    assert (xs > 0.0).all()
 50    assert (ys > 0.0).all()
 51
 52    ln_xs = xs.log()
 53    ln_ys = ys.log()
 54    coefficients = torch_polyfit(ln_xs, ln_ys, degree=1)
 55    return torch.tensor([coefficients[0].exp(), coefficients[1]])

 67def visualise_planes_as_points(grid: Sinogram3dGrid, scalars: torch.Tensor | None):
 68    ct = grid.theta.cos().flatten()
 69    st = grid.theta.sin().flatten()
 70    cp = grid.phi.cos().flatten()
 71    sp = grid.phi.sin().flatten()
 72    points = grid.r.flatten().unsqueeze(-1) * torch.stack((ct * cp, ct * sp, st), dim=-1)
 73    pl = pv.Plotter()
 74    if scalars is None:
 75        pl.add_points(points.cpu().numpy())
 76    else:
 77        pl.add_points(points.cpu().numpy(), scalars=scalars.flatten().cpu().numpy())
 78    pl.show()

260    @staticmethod
261    def _model(parameters: np.ndarray, independent_variable: np.ndarray) -> np.ndarray:
262        x_a = np.pow(independent_variable, parameters[0])
263        bx = parameters[1] * independent_variable
264        return (x_a + bx) / (2.0 - x_a + bx)
265
266    @staticmethod
267    def _function(parameters: np.ndarray, independent_variable: np.ndarray,
268                  dependent_variable: np.ndarray) -> np.ndarray:
269        return CustomFitPowerRatio._model(parameters, independent_variable) - dependent_variable
270
271    @staticmethod
272    def _jacobian(parameters: np.ndarray, independent_variable: np.ndarray,
273                  dependent_variable: np.ndarray) -> np.ndarray:
274        ret = np.empty((independent_variable.size, parameters.size))
275        x_a = np.pow(independent_variable, parameters[0])  # size matches that of independent_variable
276        bx = parameters[1] * independent_variable  # size matches that of independent_variable
277        den = 2.0 - x_a + bx  # size matches that of independent_variable
278        den = den * den  # size matches that of independent_variable
279        ret[:, 0] = 2.0 * (1.0 + bx) * x_a * np.log(independent_variable) / den
280        ret[:, 1] = 2.0 * (1.0 - x_a) * independent_variable / den
281        return ret
