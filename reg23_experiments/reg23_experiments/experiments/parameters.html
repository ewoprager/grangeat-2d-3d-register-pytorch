 79    def _update_op_algo_params(self):
 80        if self.optimisation_algorithm not in self._op_algo_cache:
 81            if self.optimisation_algorithm == "pso":
 82                self._op_algo_cache[self.optimisation_algorithm] = PsoParameters()
 83            elif self.optimisation_algorithm == "local_search":
 84                self._op_algo_cache[self.optimisation_algorithm] = LocalSearchParameters()
 85            else:
 86                raise ValueError(f"Unrecognised optimisation algorithm '{self.optimisation_algorithm}'.")
 87
 88        self.op_algo_parameters = self._op_algo_cache[self.optimisation_algorithm]
 89
 90    def _update_sim_metric_params(self):
 91        if self.sim_metric not in self._sim_metric_cache:
 92            if self.sim_metric == "zncc":
 93                self._sim_metric_cache[self.sim_metric] = NoParameters()
 94            elif self.sim_metric == "local_zncc":
 95                self._sim_metric_cache[self.sim_metric] = LocalZnccParameters()
 96            elif self.sim_metric == "multiscale_zncc":
 97                self._sim_metric_cache[self.sim_metric] = NoParameters()
 98            elif self.sim_metric == "gradient_correlation":

100            else:
101                raise ValueError(f"Unrecognised optimisation algorithm '{self.sim_metric}'.")
