 66class DAG:
 67    """
 68    A data manager that stores data in a directed, acyclic graph.
 69
 70    Each variable (piece of data) has a `str` name and is stored as a node in the graph.
 71
 72    The value of a variable can be set using `data_manager().set_data(<variable name>, <value>)`. Multiple can be set
 73    at once using `data_manager().set_data_multiple(<variable name>=<value>, <variable name>=<value>, ...)`.
 74
 75    Dependencies between variables are added using instance of the `Updater` class, and the function
 76    `data_manager().add_updater(<updater name>, <updater instance>)`. The `<updater name>` is just a user-defined `str`
 77    name associated with the updater instance, which is used if the user wishes to remove the updater later on with
 78    `data_manager().remove_updater(<updater name>)`.
 79
 80    Each updater defines a dependency of a set of variables on a set of other variables, as well as the mapping between
 81    them.
 82
 83    Each variable (node) may be updated by:
 84    - no updaters, so its value is always valid, and is only ever changed using `data_manager().set_data` /
 85    `data_manager().set_data_multiple`, or
 86    - one updater, so its value is invalidated by changes to its dependents, and will be re-evaluated automatically by
 87    its updater when read.
 88
 89    By default, all variables are evaluated lazily, i.e. updaters will only be run to re-evaluate invalidated values
 90    upon reading of the dependent variables. While this is good for optimisation, when data are displayed in a GUI, this
 91    is generally not desirable, as the user will generally wish to only be shown valid data. In this case, variables
 92    can manually be set to non-lazy have evaluation using
 93    `data_manager().set_evaluation_laziness(<variable name>, lazily_evaluated=False)`.
 94
 95    It is sometimes useful to know when a variable's value changes, e.g. for updating GUI elements. Callbacks can be
 96    added using `data_manager().add_callback(<variable name>, <callback name>, <callback value>)`. The `<callback name>`
 97    is just a user-defined `str` name associated with the callback, which is used if the user wishes to remove the
 98    callback later on with `data_manager().remove_callback(<variable name>, <callback name>)`.
 99    """

175        """
176        Set the data associated with a named node. Will create the node if it doesn't exist.
177        :param node_name: Name of the node.
178        :param data: New data to assign.
179        :param check_equality: [Optional; default=False] Whether to check the new value against the old value,

181        """

183        self.add_node(node_name)
184        # set the data and make not dirty

231    @_data_mutating
232    def add_updater(self, name: str, updater: Updater) -> None | Error:
233        """
234        Add a new updater object to the DAG. Each node may only be updated by a single updater, or no updater.
235        :param name: A human-readable name to assign to the added updater.
236        :param updater: The updater object. Can be created from a function using the `dag_updater` decorator.
237        :return: None, or an `Error` object on failure.
238        """
239        # insert the new updater, checking that there isn't already one of the same name
240        if name in self._updaters:
241            return Error(f"Updater named '{name}' already exists in graph.")
242        self._updaters[name] = updater
243        # set the updater names for updated nodes
244        for updated in updater.returned:
245            self.add_node(updated)
246            if self._nodes[updated].updater is None:
247                self._nodes[updated].updater = name
248            else:
249                return Error(
250                    f"Node {updated} is already updated by {self._nodes[updated].updater}; tried to add updater {name} "
251                    f"which wants to update the same node.")
252        # add dependencies now that updated nodes have assigned updaters
253        for dependency in updater.dependencies:
254            self._add_dependency(dependency)
255        # check that the graph is not acyclic
256        if not self._check_acyclic():
257            self.remove_updater(name)
258            return Error(f"Failed to add updater '{name}', as this added cycles to the graph.")
259        return None

261    def remove_updater(self, name: str) -> None | Error:
262        # check updater exists with given name
263        if name not in self._updaters:
264            return Error(f"Failed to remove updater '{name}'; it doesn't exist.")
265        # remove the dependencies
266        for dependency in self._updaters[name].dependencies:
267            self._remove_dependency(dependency)
268        # remove the updater names for the updated nodes
269        for updated in self._updaters[name].returned:
270            if updated not in self._nodes:
271                continue
272            self._nodes[updated].updater = None
273        # remove the updater
274        self._updaters.pop(name)
275        return None

332    def _run_updater(self, name: str, *, soft: bool = False) -> None | Error:
333        # get the updater object
334        if name not in self._updaters:
335            return Error(f"No updater named '{name}' in graph.")
336        updater = self._updaters[name]
337        # get the arguments for the function
338        args = self.get_args(updater.arguments)
339        if isinstance(args, Error):
340            if soft:
341                return None
342            return Error(f"Failed to get arguments to run updater '{name}': {args.description}")
343        # execute the function
344        res = updater.function(**args)
345        if isinstance(res, Error):
346            return Error(f"Error running updater function '{name}': {res.description}")
347        # for each promised value, check it exists, and set the data
348        if not isinstance(res, dict):
349            return Error(f"Expected data updater function '{name}' to return a dictionary.")
350        for variable_updated in updater.returned:
351            if variable_updated not in res:
352                return Error(
353                    f"Variable '{variable_updated}' not returned by updater function '{name}' which promised it.")
354            err = self.set_data(variable_updated, res.pop(variable_updated))
355            if isinstance(err, Error):
356                return Error(
357                    f"Error setting value for '{variable_updated}' after running updater '{name}': {err.description}.")
358        # check for values returned that weren't promised
359        if len(res) > 0:
360            variable_names = "', '".join(list(res.keys()))
361            return Error(f"Updater function '{name}' returned unexpected variables: '{variable_names}'")
362        return None

364    def _clean_node(self, name: str, *, soft: bool = False) -> None | Error:
365        if name not in self._nodes:
366            return Error(f"No node named '{name}' in graph.")
367        if not self._nodes[name].dirty:
368            return None
369        if self._nodes[name].updater is None:
370            if soft:
371                return None
372            return Error(f"Dirty node '{name}' cannot be cleaned as it has no updater.")
373        err = self._run_updater(self._nodes[name].updater, soft=soft)
374        if isinstance(err, Error):
375            return Error(f"Failed to clean dirty node '{name}'; error running updater '{self._nodes[name].updater}': "
376                         f"{err.description}.")
377        if self._nodes[name].dirty and not soft:
378            return Error(f"Node '{name}' still dirty after running updater '{self._nodes[name].updater}'.")
379        return None

381    def _clean_graph(self) -> None | Error:
382        for name, node in self._nodes.items():
383            if node.lazily_evaluated:
384                continue
385            err = self._clean_node(name, soft=True)
386            if isinstance(err, Error):

388        return None

390    def _eval_in_degrees(self) -> dict[str, int]:
391        ret = {node_name: 0 for node_name in self._nodes}
392        for name, node in self._nodes.items():
393            for dependent in node.dependents:

397    def _check_acyclic(self) -> bool:
398        in_degrees: dict[str, int] = self._eval_in_degrees()
399        q = collections.deque([node_name for node_name in self._nodes if in_degrees[node_name] == 0])
400        count: int = 0
401        while q:
402            first = q.popleft()
403            count += 1
404            for dependent in self._nodes[first].dependents:
405                in_degrees[dependent] -= 1
406                if in_degrees[dependent] == 0:
407                    q.append(dependent)
408        return count == len(self._nodes)

A data manager that stores data in a directed, acyclic graph.

Each variable (piece of data) has a str name and is stored as a node in the graph.

The value of a variable can be set using data_manager().set_data(<variable name>, <value>). Multiple can be set at once using data_manager().set_data_multiple(<variable name>=<value>, <variable name>=<value>, ...).

Dependencies between variables are added using instance of the Updater class, and the function data_manager().add_updater(<updater name>, <updater instance>). The <updater name> is just a user-defined str name associated with the updater instance, which is used if the user wishes to remove the updater later on with data_manager().remove_updater(<updater name>).

Each updater defines a dependency of a set of variables on a set of other variables, as well as the mapping between them.

Each variable (node) may be updated by:

  • no updaters, so its value is always valid, and is only ever changed using data_manager().set_data / data_manager().set_data_multiple, or
  • one updater, so its value is invalidated by changes to its dependents, and will be re-evaluated automatically by its updater when read.

By default, all variables are evaluated lazily, i.e. updaters will only be run to re-evaluate invalidated values upon reading of the dependent variables. While this is good for optimisation, when data are displayed in a GUI, this is generally not desirable, as the user will generally wish to only be shown valid data. In this case, variables can manually be set to non-lazy have evaluation using data_manager().set_evaluation_laziness(<variable name>, lazily_evaluated=False).

It is sometimes useful to know when a variable's value changes, e.g. for updating GUI elements. Callbacks can be added using data_manager().add_callback(<variable name>, <callback name>, <callback value>). The <callback name> is just a user-defined str name associated with the callback, which is used if the user wishes to remove the callback later on with data_manager().remove_callback(<variable name>, <callback name>).

233        """
234        Add a new updater object to the DAG. Each node may only be updated by a single updater, or no updater.
235        :param name: A human-readable name to assign to the added updater.
236        :param updater: The updater object. Can be created from a function using the `dag_updater` decorator.
237        :return: None, or an `Error` object on failure.
238        """
239        # insert the new updater, checking that there isn't already one of the same name
240        if name in self._updaters:
241            return Error(f"Updater named '{name}' already exists in graph.")
242        self._updaters[name] = updater
243        # set the updater names for updated nodes
244        for updated in updater.returned:
245            self.add_node(updated)
246            if self._nodes[updated].updater is None:
247                self._nodes[updated].updater = name
248            else:
249                return Error(
250                    f"Node {updated} is already updated by {self._nodes[updated].updater}; tried to add updater {name} "
251                    f"which wants to update the same node.")
252        # add dependencies now that updated nodes have assigned updaters
253        for dependency in updater.dependencies:
254            self._add_dependency(dependency)
255        # check that the graph is not acyclic
256        if not self._check_acyclic():
257            self.remove_updater(name)
258            return Error(f"Failed to add updater '{name}', as this added cycles to the graph.")
259        return None
