106 __host__ __device__ static float ResamplePlane(const sinogram_t &sinogram, const ConstantGeometry &geometry,
107 float phi, float r) {
108 const float cp = cosf(phi);
109 const float sp = sinf(phi);
110 const Vec<float, 4> intermediate = MatMul(geometry.projectionMatrixTranspose, Vec<float, 4>{cp, sp, 0.f, -r});
111 const Vec<float, 3> intermediate3 = intermediate.XYZ();
112 const Vec<float, 3> posCartesian =
113 -intermediate.W() * intermediate3 / intermediate3.Apply<float>(&Square<float>).Sum();
114
115 Vec<double, 3> rThetaPhi{};
116 rThetaPhi.Z() = atan2(posCartesian.Y(), posCartesian.X());
117 const float magXY = posCartesian.X() * posCartesian.X() + posCartesian.Y() * posCartesian.Y();
118 rThetaPhi.Y() = atan2(posCartesian.Z(), sqrt(magXY));
119 rThetaPhi.X() = sqrt(magXY + posCartesian.Z() * posCartesian.Z());
120 rThetaPhi = UnflipSphericalCoordinate(rThetaPhi);
121
122 float ret = sinogram.Sample(rThetaPhi);
123
124 if ((r * Vec<float, 2>{cp, sp} - .5f * geometry.originProjection).Apply<float>(&Square<float>).Sum() <
125 geometry.squareRadius) {
126 ret *= -1.f;
127 }
128 return ret;
129 }
