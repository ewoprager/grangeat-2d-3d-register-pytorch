64 [[nodiscard]] __host__ __device__ float Sample(VectorType texCoord) const {
65 texCoord = texCoord * Size().StaticCast<double>() - .5;
66 const VectorType floored = texCoord.Apply<double>(&floor);
67 const SizeType index = floored.StaticCast<int64_t>();
68 const VectorType fractions = texCoord - floored;
69 const float l0r0 =
70 (1.f - fractions.X()) * At(index) + fractions.X() * At({index.X() + 1, index.Y(), index.Z()});
71 const float l0r1 = (1.f - fractions.X()) * At({index.X(), index.Y() + 1, index.Z()}) +
72 fractions.X() * At({index.X() + 1, index.Y() + 1, index.Z()});
73 const float l1r0 = (1.f - fractions.X()) * At({index.X(), index.Y(), index.Z() + 1}) +
74 fractions.X() * At({index.X() + 1, index.Y(), index.Z() + 1});
75 const float l1r1 = (1.f - fractions.X()) * At({index.X(), index.Y() + 1, index.Z() + 1}) +
76 fractions.X() * At({index.X() + 1, index.Y() + 1, index.Z() + 1});
77 const float l0 = (1.f - fractions.Y()) * l0r0 + fractions.Y() * l0r1;
78 const float l1 = (1.f - fractions.Y()) * l1r0 + fractions.Y() * l1r1;
79 return (1.f - fractions.Z()) * l0 + fractions.Z() * l1;
80 }

87 [[nodiscard]] __host__ __device__ float DSampleDX(VectorType texCoord) const {
88 const VectorType sizeF = Size().StaticCast<double>();
89 texCoord = texCoord * sizeF - .5;
90 const VectorType floored = texCoord.Apply<double>(&floor);
91 const SizeType index = floored.StaticCast<int64_t>();
92 const float fVertical = texCoord.Y() - floored.Y();
93 const float fInward = texCoord.Z() - floored.Z();
94 const float l0 =
95 (1.f - fVertical) * (At({index.X() + 1, index.Y(), index.Z()}) - At(index)) +
96 fVertical * (At({index.X() + 1, index.Y() + 1, index.Z()}) - At({index.X(), index.Y() + 1, index.Z()}));
97 const float l1 =
98 (1.f - fVertical) *
99 (At({index.X() + 1, index.Y(), index.Z() + 1}) - At({index.X(), index.Y(), index.Z() + 1})) +
100 fVertical *
101 (At({index.X() + 1, index.Y() + 1, index.Z() + 1}) - At({index.X(), index.Y() + 1, index.Z() + 1}));
102 return sizeF.X() * ((1.f - fInward) * l0 + fInward * l1);
103 }
