105 __host__ static CommonData Common(const at::Tensor &volume, const at::Tensor &volumeSpacing,
106 const at::Tensor &phiValues, const at::Tensor &thetaValues,
107 const at::Tensor &rValues, int64_t samplesPerDirection, at::DeviceType device) {
108 // volume should be a 3D array of floats on the chosen device
109 TORCH_CHECK(volume.sizes().size() == 3);
110 TORCH_CHECK(volume.dtype() == at::kFloat);
111 TORCH_INTERNAL_ASSERT(volume.device().type() == device);
112 // volumeSpacing should be a 1D tensor of 3 doubles
113 TORCH_CHECK(volumeSpacing.sizes() == at::IntArrayRef{3});
114 TORCH_CHECK(volumeSpacing.dtype() == at::kDouble);
115 // phiValues, thetaValues and rValues should have matching sizes, contain floats, and be on the chosen device
116 TORCH_CHECK(phiValues.sizes() == thetaValues.sizes());
117 TORCH_CHECK(thetaValues.sizes() == rValues.sizes());
118 TORCH_CHECK(phiValues.dtype() == at::kFloat);
119 TORCH_CHECK(thetaValues.dtype() == at::kFloat);
120 TORCH_CHECK(rValues.dtype() == at::kFloat);
121 TORCH_INTERNAL_ASSERT(phiValues.device().type() == device);
122 TORCH_INTERNAL_ASSERT(thetaValues.device().type() == device);
123 TORCH_INTERNAL_ASSERT(rValues.device().type() == device);
124
125 CommonData ret{};
126 ret.inputTexture = texture_t::FromTensor(volume, volumeSpacing);
127 const double planeSize = sqrtf(ret.inputTexture.SizeWorld().template Apply<double>(&Square<double>).Sum());
128 ret.mappingIndexToOffset = GetMappingIToOffset(planeSize, samplesPerDirection);
129 const double rootScaleFactor = planeSize / static_cast<float>(samplesPerDirection);
130 ret.scaleFactor = rootScaleFactor * rootScaleFactor;
131 ret.flatOutput = torch::zeros(at::IntArrayRef({phiValues.numel()}), volume.contiguous().options());
132 return ret;
133 }

164 __host__ __device__ [[nodiscard]] static Linear2<Vec<double, 3> >
165 GetMappingIndexToTexCoord(const texture_t &textureIn, double phi, double theta, double r,
166 const Linear<Vec<double, 3> > &mappingIToOffset) {
167 const double sp = sin(phi);
168 const double cp = cos(phi);
169 const double st = sin(theta);
170 const double ct = cos(theta);
171 const Linear2<Vec<double, 3> > mappingOffsetToWorld{{r * ct * cp, r * ct * sp, r * st}, {-sp, cp, 0.f},
172 {-st * cp, -st * sp, ct}};
173 return textureIn.MappingWorldToTexCoord()(mappingOffsetToWorld(mappingIToOffset));
174 }

188 __host__ __device__ [[nodiscard]] static Vec<double, 3> GetDTexCoordDR(const texture_t &textureIn, double phi,
189 double theta, double r) {
190 const double sign = static_cast<double>(r > 0.) - static_cast<double>(r < 0.);
191 const double sp = sin(phi);
192 const double cp = cos(phi);
193 const double st = sin(theta);
194 const double ct = cos(theta);
195 return textureIn.MappingWorldToTexCoord().gradient * sign * Vec<double, 3>{ct * cp, ct * sp, st};
196 }

211 __host__ __device__ [[nodiscard]] static float
212 IntegrateLooped(const texture_t &texture, const Linear2<Vec<double, 3> > &mappingIndexToTexCoord,
213 int64_t samplesPerDirection) {
214 float ret = 0.f;
215 for (int64_t j = 0; j < samplesPerDirection; ++j) {
216 for (int64_t i = 0; i < samplesPerDirection; ++i) {
217 const double iF = static_cast<double>(i);
218 const double jF = static_cast<double>(j);
219 ret += texture.Sample(mappingIndexToTexCoord(Vec<double, 3>::Full(iF), Vec<double, 3>::Full(jF)));
220 }
221 }
222 return ret;
223 }

236 DIntegrateLoopedDMappingParameter(const texture_t &texture, const Linear2<Vec<double, 3> > &mappingIndexToTexCoord,
237 const Vec<double, 3> &dTexCoordDMappingParameter, int64_t samplesPerDirection) {
238 float ret = 0.f;
239 for (int64_t j = 0; j < samplesPerDirection; ++j) {
240 for (int64_t i = 0; i < samplesPerDirection; ++i) {
241 const double iF = static_cast<double>(i);
242 const double jF = static_cast<double>(j);
243 const Vec<double, 3> texCoord = mappingIndexToTexCoord(Vec<double, 3>::Full(iF),
244 Vec<double, 3>::Full(jF));
245 ret += texture.DSampleDX(texCoord) * dTexCoordDMappingParameter.X() + texture.DSampleDY(texCoord) *
246 dTexCoordDMappingParameter.Y() + texture.DSampleDZ(texCoord) * dTexCoordDMappingParameter.Z();
247 }
248 }
