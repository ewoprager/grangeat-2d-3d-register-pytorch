95 __host__ static CommonData Common(const at::Tensor &volume, const at::Tensor &volumeSpacing,
96 const at::Tensor &phiValues, const at::Tensor &thetaValues,
97 const at::Tensor &rValues, int64_t samplesPerDirection, at::DeviceType device) {
98 // volume should be a 3D array of floats on the chosen device
99 TORCH_CHECK(volume.sizes().size() == 3);
100 TORCH_CHECK(volume.dtype() == at::kFloat);
101 TORCH_INTERNAL_ASSERT(volume.device().type() == device);
102 // volumeSpacing should be a 1D tensor of 3 floats or doubles
103 TORCH_CHECK(volumeSpacing.sizes() == at::IntArrayRef{3});
104 TORCH_CHECK(volumeSpacing.dtype() == at::kFloat || volumeSpacing.dtype() == at::kDouble);
105 // phiValues, thetaValues and rValues should have matching sizes, contain floats, and be on the chosen device
106 TORCH_CHECK(phiValues.sizes() == thetaValues.sizes());
107 TORCH_CHECK(thetaValues.sizes() == rValues.sizes());
108 TORCH_CHECK(phiValues.dtype() == at::kFloat);
109 TORCH_CHECK(thetaValues.dtype() == at::kFloat);
110 TORCH_CHECK(rValues.dtype() == at::kFloat);
111 TORCH_INTERNAL_ASSERT(phiValues.device().type() == device);
112 TORCH_INTERNAL_ASSERT(thetaValues.device().type() == device);
113 TORCH_INTERNAL_ASSERT(rValues.device().type() == device);
114
115 CommonData ret{};
116 ret.inputTexture = texture_t::FromTensor(volume, volumeSpacing);
117 const double planeSize = sqrtf(ret.inputTexture.SizeWorld().template Apply<double>(&Square<double>).Sum());
118 ret.mappingIndexToOffset = GetMappingIToOffset(planeSize, samplesPerDirection);
119 const double rootScaleFactor = planeSize / static_cast<float>(samplesPerDirection);
120 ret.scaleFactor = rootScaleFactor * rootScaleFactor;
121 ret.flatOutput = torch::zeros(at::IntArrayRef({phiValues.numel()}), volume.contiguous().options());
122 return ret;
123 }

154 __host__ __device__ [[nodiscard]] static Linear2<Vec<double, 3>>
155 GetMappingIndexToTexCoord(const texture_t &textureIn, double phi, double theta, double r,
156 const Linear<Vec<double, 3>> &mappingIToOffset) {
157 const double sp = sin(phi);
158 const double cp = cos(phi);
159 const double st = sin(theta);
160 const double ct = cos(theta);
161 const Linear2<Vec<double, 3>> mappingOffsetToWorld{
162 {r * ct * cp, r * ct * sp, r * st}, {-sp, cp, 0.f}, {-st * cp, -st * sp, ct}};
163 return textureIn.MappingWorldToTexCoord()(mappingOffsetToWorld(mappingIToOffset));
164 }

178 __host__ __device__ [[nodiscard]] static Vec<double, 3> GetDTexCoordDR(const texture_t &textureIn, double phi,
179 double theta, double r) {
180 const double sign = static_cast<double>(r > 0.) - static_cast<double>(r < 0.);
181 const double sp = sin(phi);
182 const double cp = cos(phi);
183 const double st = sin(theta);
184 const double ct = cos(theta);
185 return textureIn.MappingWorldToTexCoord().gradient * sign * Vec<double, 3>{ct * cp, ct * sp, st};
186 }

201 __host__ __device__ [[nodiscard]] static float
202 IntegrateLooped(const texture_t &texture, const Linear2<Vec<double, 3>> &mappingIndexToTexCoord,
203 int64_t samplesPerDirection) {
204 float ret = 0.f;
205 for (int64_t j = 0; j < samplesPerDirection; ++j) {
206 for (int64_t i = 0; i < samplesPerDirection; ++i) {
207 const double iF = static_cast<double>(i);
208 const double jF = static_cast<double>(j);
209 ret += texture.Sample(mappingIndexToTexCoord(Vec<double, 3>::Full(iF), Vec<double, 3>::Full(jF)));
210 }
211 }
212 return ret;
213 }

226 DIntegrateLoopedDMappingParameter(const texture_t &texture, const Linear2<Vec<double, 3>> &mappingIndexToTexCoord,
227 const Vec<double, 3> &dTexCoordDMappingParameter, int64_t samplesPerDirection) {
228 float ret = 0.f;
229 for (int64_t j = 0; j < samplesPerDirection; ++j) {
230 for (int64_t i = 0; i < samplesPerDirection; ++i) {
231 const double iF = static_cast<double>(i);
232 const double jF = static_cast<double>(j);
233 const Vec<double, 3> texCoord =
234 mappingIndexToTexCoord(Vec<double, 3>::Full(iF), Vec<double, 3>::Full(jF));
235 ret += texture.DSampleDX(texCoord) * dTexCoordDMappingParameter.X() +
236 texture.DSampleDY(texCoord) * dTexCoordDMappingParameter.Y() +
237 texture.DSampleDZ(texCoord) * dTexCoordDMappingParameter.Z();
238 }
239 }
240 return ret;
