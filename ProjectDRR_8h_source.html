67 const at::Tensor &detectorSpacing, int64_t samplesPerRay, at::DeviceType device) {
68 // volume should be a 3D tensor of floats on the chosen device
69 TORCH_CHECK(volume.sizes().size() == 3);
70 TORCH_CHECK(volume.dtype() == at::kFloat);

74 TORCH_CHECK(voxelSpacing.dtype() == at::kDouble);
75 // homographyMatrixInverse should be of size (4, 4), contain doubles and be on the chosen device
76 TORCH_CHECK(homographyMatrixInverse.sizes() == at::IntArrayRef({4, 4}));
77 TORCH_CHECK(homographyMatrixInverse.dtype() == at::kDouble);
78 TORCH_INTERNAL_ASSERT(homographyMatrixInverse.device().type() == device);
79 // outputOffset should be a 1D tensor of 2 doubles
80 TORCH_CHECK(outputOffset.sizes() == at::IntArrayRef{2});
81 TORCH_CHECK(outputOffset.dtype() == at::kDouble);
82 // detectorSpacing should be a 1D tensor of 2 doubles
83 TORCH_CHECK(detectorSpacing.sizes() == at::IntArrayRef{2});
84 TORCH_CHECK(detectorSpacing.dtype() == at::kDouble);
85
86 CommonData ret{};
87 ret.inputTexture = texture_t::FromTensor(volume, voxelSpacing);
88 ret.homographyMatrixInverse = Vec<Vec<double, 4>, 4>::FromTensor2D(homographyMatrixInverse);
89
90 const Vec<int64_t, 3> inputSize = Vec<int64_t, 3>::FromIntArrayRef(volume.sizes()).Flipped();
91 const VectorType volumeDiagonal = inputSize.StaticCast<FloatType>() * ret.inputTexture.Spacing();
92 const FloatType volumeDiagLength = volumeDiagonal.Length();
93 const VectorType sourcePosition = {0.0, 0.0, sourceDistance};
94 ret.lambdaStart = MatMul(ret.homographyMatrixInverse, VecCat(sourcePosition, 1.0)).XYZ().Length() - 0.5 *
95 volumeDiagLength;
96 ret.stepSize = volumeDiagLength / static_cast<FloatType>(samplesPerRay);
97 ret.outputOffset = Vec<double, 2>::FromTensor(outputOffset);
98 ret.detectorSpacing = Vec<double, 2>::FromTensor(detectorSpacing);
99 ret.flatOutput = torch::zeros(at::IntArrayRef({outputWidth * outputHeight}), volume.contiguous().options());
