46 __host__ static CommonData Common(const at::Tensor &volume, const at::Tensor &voxelSpacing,

48 int64_t outputWidth, int64_t outputHeight, const at::Tensor &detectorSpacing,
49 int64_t samplesPerRay, at::DeviceType device) {
50 // volume should be a 3D tensor of floats on the chosen device
51 TORCH_CHECK(volume.sizes().size() == 3);
52 TORCH_CHECK(volume.dtype() == at::kFloat);
53 TORCH_INTERNAL_ASSERT(volume.device().type() == device);
54 // voxelSpacing should be a 1D tensor of 3 doubles
55 TORCH_CHECK(voxelSpacing.sizes() == at::IntArrayRef{3});
56 TORCH_CHECK(voxelSpacing.dtype() == at::kDouble);
57 // homographyMatrixInverse should be of size (4, 4), contain doubles and be on the chosen device
58 TORCH_CHECK(homographyMatrixInverse.sizes() == at::IntArrayRef({4, 4}));
59 TORCH_CHECK(homographyMatrixInverse.dtype() == at::kDouble);
60 TORCH_INTERNAL_ASSERT(homographyMatrixInverse.device().type() == device);
61 // detectorSpacing should be a 1D tensor of 2 doubles
62 TORCH_CHECK(detectorSpacing.sizes() == at::IntArrayRef{2});
63 TORCH_CHECK(detectorSpacing.dtype() == at::kDouble);
64
65 CommonData ret{};
66 ret.inputTexture = texture_t::FromTensor(volume, voxelSpacing);
67 ret.homographyMatrixInverse = Vec<Vec<double, 4>, 4>::FromTensor2D(homographyMatrixInverse);
68
69 const Vec<int64_t, 3> inputSize = Vec<int64_t, 3>::FromIntArrayRef(volume.sizes()).Flipped();
70 const Vec<double, 3> volumeDiagonal = inputSize.StaticCast<double>() * ret.inputTexture.Spacing();
71 const double volumeDiagLength = volumeDiagonal.Length();
72 const Vec<double, 3> sourcePosition = {0.0, 0.0, sourceDistance};
73 ret.lambdaStart = MatMul(ret.homographyMatrixInverse, VecCat(sourcePosition, 1.0)).XYZ().Length() - 0.5 *
74 volumeDiagLength;
75 ret.stepSize = volumeDiagLength / static_cast<double>(samplesPerRay);
76
77 ret.detectorSpacing = Vec<double, 2>::FromTensor(detectorSpacing);
78 ret.flatOutput = torch::zeros(at::IntArrayRef({outputWidth * outputHeight}), volume.contiguous().options());
79 return ret;
