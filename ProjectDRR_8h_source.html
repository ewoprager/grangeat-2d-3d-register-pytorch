67 __host__ static CommonData Common(const at::Tensor &volume, const at::Tensor &voxelSpacing,
68 const at::Tensor &homographyMatrixInverse, double sourceDistance,
69 int64_t outputWidth, int64_t outputHeight, const at::Tensor &outputOffset,
70 const at::Tensor &detectorSpacing, at::DeviceType device,
71 std::optional<int64_t> samplesPerRay = std::nullopt) {
72 // volume should be a 3D tensor of floats on the chosen device
73 TORCH_CHECK(volume.sizes().size() == 3);
74 TORCH_CHECK(volume.dtype() == at::kFloat);
75 TORCH_INTERNAL_ASSERT(volume.device().type() == device);
76 // voxelSpacing should be a 1D tensor of 3 doubles
77 TORCH_CHECK(voxelSpacing.sizes() == at::IntArrayRef{3});
78 TORCH_CHECK(voxelSpacing.dtype() == at::kDouble);
79 // homographyMatrixInverse should be of size (4, 4), contain doubles and be on the chosen device
80 TORCH_CHECK(homographyMatrixInverse.sizes() == at::IntArrayRef({4, 4}));
81 TORCH_CHECK(homographyMatrixInverse.dtype() == at::kDouble);
82 TORCH_INTERNAL_ASSERT(homographyMatrixInverse.device().type() == device);
83 // outputOffset should be a 1D tensor of 2 doubles
84 TORCH_CHECK(outputOffset.sizes() == at::IntArrayRef{2});
85 TORCH_CHECK(outputOffset.dtype() == at::kDouble);
86 // detectorSpacing should be a 1D tensor of 2 doubles
87 TORCH_CHECK(detectorSpacing.sizes() == at::IntArrayRef{2});
88 TORCH_CHECK(detectorSpacing.dtype() == at::kDouble);
89
90 const int64_t samplesPerRayValue = samplesPerRay.value_or(
91 Vec<int64_t, 3>::FromIntArrayRef(volume.sizes()).Max());
92
93 CommonData ret{};
94 ret.inputTexture = texture_t::FromTensor(volume, VectorType::FromTensor(voxelSpacing));
95 ret.homographyMatrixInverse = Vec<Vec<double, 4>, 4>::FromTensor2D(homographyMatrixInverse);
96
97 const Vec<int64_t, 3> inputSize = Vec<int64_t, 3>::FromIntArrayRef(volume.sizes()).Flipped();
98 const VectorType volumeDiagonal = inputSize.StaticCast<FloatType>() * ret.inputTexture.Spacing();
99 const FloatType volumeDiagLength = volumeDiagonal.Length();
100 const VectorType sourcePosition = {0.0, 0.0, sourceDistance};
101 ret.lambdaStart = MatMul(ret.homographyMatrixInverse, VecCat(sourcePosition, 1.0)).XYZ().Length() - 0.5 *
102 volumeDiagLength;
103 ret.stepSize = volumeDiagLength / static_cast<FloatType>(samplesPerRayValue);
104 ret.samplesPerRay = samplesPerRayValue;
105 ret.outputOffset = Vec<double, 2>::FromTensor(outputOffset);
106 ret.detectorSpacing = Vec<double, 2>::FromTensor(detectorSpacing);
107 ret.flatOutput = torch::zeros(at::IntArrayRef({outputWidth * outputHeight}), volume.contiguous().options());
108 return ret;
109 }
