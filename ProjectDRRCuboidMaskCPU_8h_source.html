52template <typename T, std::size_t faceCount> __host__ __device__ T RayConvexPolyhedronDistance(
53 const std::array<Vec<T, 3>, faceCount> &facePoints, const std::array<Vec<T, 3>, faceCount> &faceOutUnitNormals,
54 const Vec<T, 3> &rayPoint, const Vec<T, 3> &rayUnitDirection) {
55
56 constexpr T epsilon = 1e-8;
57
58 T entryLambda = -std::numeric_limits<T>::infinity();
59 T exitLambda = std::numeric_limits<T>::infinity();
60 std::size_t entryIndex = 0;
61 for (std::size_t i = 0; i < faceCount; ++i) {
62 const T dot = VecDot(faceOutUnitNormals[i], rayUnitDirection);
63 const T lambda = VecDot(faceOutUnitNormals[i], facePoints[i] - rayPoint) / dot;
64 // If dot is 0, the ray is parallel to the face; if dot is negative, increasing lambda moves you along the ray
65 // in the 'inward' direction relative to this face, and vice versa if dot is positive.
66 if (dot < -epsilon && lambda > entryLambda) {
67 // Moving in 'inward' direction, so could be entering the polyhedron through this face
68 entryLambda = lambda;
69 entryIndex = i;
70 } else if (dot > epsilon && lambda < exitLambda) {
71 // Moving in 'outward' direction, so could be exiting the polyhedron through this face
72 exitLambda = lambda;
73 }
74 }
75 // `entryLambda` is now the largest value for which ray is moving 'into' face, and `entryIndex` is index of
76 // corresponding face. `exitLambda` is the smallest value for which ray is moving 'out of' face. This means that,
77 // **if the ray intersects with the polyhedron**, it will enter at `entryLambda` and leave at `exitLambda`.
78
79 const Vec<T, 3> entryPoint = rayPoint + entryLambda * rayUnitDirection;
80 // Checking that the entry point lies 'within' all the other faces of the polyhedron. If it lies 'outside' of any,
81 // it doesn't intersect the polyhedron. Technically, only the faces adjacent to the intersected face need to be
82 // checked, but that would require expensive analysis of the face intersections.
83 for (std::size_t i = 0; i < faceCount; ++i) {
84 if (i == entryIndex) continue;
85 if (VecDot(entryPoint - facePoints[i], faceOutUnitNormals[i]) > epsilon) return 0.f;
86 }
87
88 return std::abs(exitLambda - entryLambda);
89}
