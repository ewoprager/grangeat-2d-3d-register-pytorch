66 [[nodiscard]] __host__ __device__ float Sample(const VectorType &rThetaPhi) const {
67 const FloatType nSideF = static_cast<FloatType>(nSide);
68
69 // to x_s, y_s
70 const FloatType z = sin(rThetaPhi[1]);
71 const FloatType zAbs = abs(z);
72 const bool equatorialZone = zAbs <= 2.0 / 3.0;
73 const FloatType sigma = Sign(z) * (2.0 - sqrt(3.0 * (1.0 - zAbs)));
74 const FloatType xS = equatorialZone ? rThetaPhi[2] + 0.5 * M_PI // with pi/2 adjustment
75 : // polar cap
76 rThetaPhi[2] + 0.5 * M_PI -
77 (abs(sigma) - 1.0) * (fmod(rThetaPhi[2] + 0.5 * M_PI, 0.5 * M_PI) -
78 0.25 * M_PI); // with pi/2 adjustment
79 const FloatType yS = equatorialZone ? 3.0 * M_PI * z / 8.0 : // polar cap
80 M_PI * sigma / 4.0;
81
82 // to x_p, y_p
83 const FloatType xP = 2.0 * nSideF * xS / M_PI;
84 const FloatType yP = nSideF * (1.0 - 2.0 * yS / M_PI);
85
86 // to u, v
87 FloatType u = xP - yP + 1.5 * nSideF;
88 FloatType v = xP + yP - 0.5 * nSideF;
89 const bool vHigh = v >= 2.0 * nSideF;
90 const bool uHigh = u >= 2.0 * nSideF;
91 if (vHigh) {
92 v -= 2.0 * nSideF;
93 if (uHigh)
94 u -= 2.0 * nSideF;
95 else {
96 u += nSideF + 2.0; // the 2 adjusts for padding
97 v -= 2.0; // this adjusts for padding
98 }
