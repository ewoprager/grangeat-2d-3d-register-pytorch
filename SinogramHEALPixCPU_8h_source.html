64 [[nodiscard]] __host__ __device__ float Sample(VectorType rThetaPhi) const {
65 const FloatType z = sin(rThetaPhi.Y());
66 const FloatType nSideF = static_cast<FloatType>(nSide);
67 const FloatType i = nSideF * (2.0 - 1.5 * z);
68 FloatType jEquiv;
69 if (z > 0.666666666666666667) {
70 // HEALPix top polar cap
71 const FloatType j = 2.0 * i * rThetaPhi.Z() / M_PI + 0.5;
72 jEquiv = nSideF * floor((j - 1.0) / i) + 1.0 + floor(0.5 * (nSideF - 1.0)) + fmod(j - 1.0, i);
73 } else if (z > -0.66666666666667) {
74 // HEALPix equatorial zone
75 const FloatType j = 2.0 * nSideF * rThetaPhi.Z() / M_PI + 0.5 * fmod(i - nSideF + 1.0, 2.0);
76 jEquiv = j;
77 } else {
78 // HEALPix bottom polar cap
79 const FloatType j = 2.0 * (4.0 * nSideF - i) * rThetaPhi.Z() / M_PI + 0.5;
80 jEquiv = nSideF * floor((j - 1.0) / (4.0 * nSideF - i)) + 1.0 + floor(0.5 * (nSideF - 1.0)) + fmod(
81 j - 1.0, 4.0 * nSideF - i);
82 }
83 const FloatType deltaV48 = jEquiv - floor(0.5 * i) + 1.0 < 0.0 ? nSideF : 0.0;
84 // r,u,v is the order of the texture dimensions
85 const VectorType ruv = {rThetaPhi.X(), fmod(jEquiv - floor(0.5 * i) + 1.0, 4.0 * nSideF),
86 fmod(jEquiv + floor(0.5 * (i + 1.0)) - 3.0 + deltaV48, 3.0 * nSideF)};
