63 // input should be a 3D tensor of floats on the chosen device
64 TORCH_CHECK(input.sizes().size() == 3)
65 TORCH_CHECK(input.dtype() == at::kFloat)
66 TORCH_INTERNAL_ASSERT(input.device().type() == device)

68 TORCH_CHECK(grid.sizes().back() == 3);
69 TORCH_CHECK(grid.dtype() == at::kFloat);

71
72 // All addressMode<dim>s should be one of the valid values:
73 AddressModeType addressModes = AddressModeType::Full(TextureAddressMode::ZERO);
74 int dim = 0;
75 for (const std::string_view &str : std::array<std::string_view, 3>
76 {{addressModeX, addressModeY, addressModeZ}}) {
77 if (str == "wrap") {
78 addressModes[dim] = TextureAddressMode::WRAP;
79 } else if (str != "zero") {
80 TORCH_WARN(
81 "Invalid address mode string given. Valid values are: 'zero', 'wrap'. Using default value: 'zero'.")
82 }
83 ++dim;
84 }

86 CommonData ret{};
87 const at::Tensor inputContiguous = input.contiguous();
88 const float *inputPtr = inputContiguous.data_ptr<float>();
89 const SizeType inputSize = SizeType::FromIntArrayRef(input.sizes()).Flipped();
90 const VectorType inputSpacing = 2.0 / (inputSize - static_cast<IntType>(1)).template StaticCast<FloatType>();
91 ret.inputTexture = texture_t{inputPtr, inputSize, inputSpacing, VectorType::Full(0.0), addressModes};
