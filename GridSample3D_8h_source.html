46 TORCH_CHECK(input.dtype() == at::kFloat)
47 TORCH_INTERNAL_ASSERT(input.device().type() == device)
48 // grid should be a tensor of floats with a final dimension of 3 on the chosen device
49 TORCH_CHECK(grid.sizes().back() == 3);
50 TORCH_CHECK(grid.dtype() == at::kFloat);
51 TORCH_INTERNAL_ASSERT(grid.device().type() == device)
52 // addressMode should be one of the valid values:
53 TextureAddressMode am = TextureAddressMode::ZERO;
54 if (addressMode == "wrap") {
55 am = TextureAddressMode::WRAP;
56 } else if (addressMode != "zero") {
57 TORCH_WARN(
58 "Invalid address mode string given. Valid values are: 'zero', 'wrap'. Using default value: 'zero'.")
59 }
60
61 CommonData ret{};
62 const at::Tensor inputContiguous = input.contiguous();
63 const float *inputPtr = inputContiguous.data_ptr<float>();
64 const Vec<int64_t, 3> inputSize = Vec<int64_t, 3>::FromIntArrayRef(input.sizes()).Flipped();
65 const Vec<double, 3> inputSpacing = 2.0 / (inputSize - static_cast<int64_t>(1)).StaticCast<double>();
66 ret.inputTexture = texture_t{inputPtr, inputSize, inputSpacing, Vec<double, 3>::Full(0.0),
67 texture_t::AddressModeType::Full(am)};
68 ret.flatOutput = torch::zeros(at::IntArrayRef({grid.numel() / 3}), inputContiguous.options());
