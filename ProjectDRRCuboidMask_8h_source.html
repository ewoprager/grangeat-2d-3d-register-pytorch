22template <typename T, std::size_t faceCount> __host__ __device__ T RayConvexPolyhedronDistance(
23 const std::array<Vec<T, 3>, faceCount> &facePoints, const std::array<Vec<T, 3>, faceCount> &faceOutUnitNormals,
24 const Vec<T, 3> &rayPoint, const Vec<T, 3> &rayUnitDirection) {
25
26 constexpr T epsilon = 1e-8;
27
28 T entryLambda = -std::numeric_limits<T>::infinity();
29 T exitLambda = std::numeric_limits<T>::infinity();
30 int entryIndex = 0;
31 for (int i = 0; i < faceCount; ++i) {
32 const T dot = VecDot(faceOutUnitNormals[i], rayUnitDirection);
33 const T lambda = VecDot(faceOutUnitNormals[i], facePoints[i] - rayPoint) / dot;
34 if (dot < -epsilon && lambda > entryLambda) {
35 entryLambda = lambda;
36 entryIndex = i;
37 } else if (dot > epsilon && lambda < exitLambda) {
38 exitLambda = lambda;
39 }
40 }
41
42 const Vec<T, 3> entryPoint = rayPoint + entryLambda * rayUnitDirection;
43 for (int i = 0; i < facePoints.size(); ++i) {
44 if (i == entryIndex) continue;
45 if (VecDot(entryPoint - facePoints[i], faceOutUnitNormals[i]) > epsilon) return 0.f;
46 }
47
48 return std::abs(exitLambda - entryLambda);
49}
